    loss/.style={rectangle, rounded corners, draw=red!70!black, thick, fill=red!15, minimum height=1.0cm, text width=2.4cm, align=center, font=\small\bfseries, outer sep=2pt},
    flow/.style={-{Stealth[length=2mm]}, thick, black!80},
    flowdash/.style={-{Stealth[length=2mm]}, thick, dashed, black!60},
    >=Stealth
]

% ==================== NODES ====================
% Input
\node (input) [data, fill=blue!15] at (0, 5.5) {Light Field Input\\$I(u,v)$, $9 \times 9$};

% Core Innovation 1: Decomposition
\node (decomp) [module, fill=orange!25] at (3.8, 5.5) {Three-Layer Angular\\Signal Decomposition};

% Intermediate Features
\node (feat_disp) [feat, fill=yellow!20] at (6.2, 7.3) {DC, $a$, $b$\\(Disparity)};
\node (feat_mat) [feat, fill=yellow!20] at (6.2, 5.5) {Residual $\epsilon(u,v)$\\(Material)};
\node (feat_grad) [feat, fill=yellow!20] at (6.2, 3.7) {Angular\\Gradient};

% Branches
\node (lamb) [module, fill=blue!15] at (8.8, 7.3) {Lambertian EPI\\Branch};
\node (nlamb) [module, fill=green!15] at (8.8, 3.7) {Non-Lambertian\\Geometric Branch};

% Outputs

\node (lossD) [loss] at (15.5, 8.7) {Depth Loss\\(MAE)};
\node (lossM) [loss] at (15.5, 2.3) {Mask Loss\\(BCE)};
\node (pseudo) [feat, fill=red!10, draw=red!50] at (12.2, 2.3) {Pseudo Label\\(Angular Grad.)};

% ==================== BOUNDING BOXES ====================
\node[draw=black!40, dashed, thick, rounded corners, fit=(lamb) (nlamb), inner sep=12pt, label={[font=\small\bfseries, black!60]above:Dual-Branch Depth Extraction}] (branch_box) {};

% ==================== CONNECTIONS ====================
% 1. Input to Decomposition
\draw[flow] (input.east) -- node[above, font=\scriptsize, pos=0.5] {$I(u,v)$} (decomp.west);

% 2. Decomposition to Features
\coordinate (decomp_out) at ($(decomp.east) + (0.5, 0)$);
\draw[flow] (decomp.east) -- (decomp_out);
\draw[flow] (decomp_out) |- (feat_disp.west);
\draw[flow] (decomp_out) -- (feat_mat.west);
\draw[flow] (decomp_out) |- (feat_grad.west);

% 3. Features to Branches
\draw[flow] (feat_disp.east) -- (lamb.west);
\coordinate (nlamb_in) at ($(nlamb.west) + (-0.5, 0)$);
\draw[flow] (feat_mat.east) -- ++(0.4, 0) |- (nlamb_in);
\draw[flow] (feat_grad.east) -- ++(0.4, 0) |- (nlamb_in);
\draw[flow] (nlamb_in) -- (nlamb.west);

% 4. Branches to Fusion
\coordinate (fuse_in_top) at ($(fuse.west) + (-0.6, 0.5)$);
\coordinate (fuse_in_bot) at ($(fuse.west) + (-0.6, -0.5)$);
\coordinate (lamb_out) at ($(lamb.east) + (0.6, 0)$);
\coordinate (nlamb_out) at ($(nlamb.east) + (0.6, 0)$);

\draw[flow] (lamb.east) -- (lamb_out);
\draw[flow] (lamb_out) |- node[pos=0.75, above, font=\scriptsize] {$D_L$} (fuse_in_top);
\draw[flow] (fuse_in_top) -- (fuse.west);

\draw[flow] (nlamb.east) -- (nlamb_out);
\draw[flow] (nlamb_out) |- node[pos=0.75, above, font=\scriptsize] {$D_{NL}$} (fuse_in_bot);
\draw[flow] (fuse_in_bot) -- (fuse.west);

% 5. Angular Gradient to Fusion (Mask Generation Supervision/Feature)
\coordinate (grad_down) at ($(feat_grad.south) + (0, -0.4)$);
\draw[flowdash] (feat_grad.south) -- (grad_down);
\draw[flowdash] (grad_down) -- ++(0, -0.8) -| node[pos=0.25, right, font=\scriptsize] {Pseudo} (fuse.south);

% 6. Fusion to Outputs
\coordinate (fuse_out) at ($(fuse.east) + (0.6, 0)$);
\draw[flow] (fuse.east) -- (fuse_out);
\draw[flow] (fuse_out) |- (outD.west);
\draw[flow] (fuse_out) |- (outM.west);

% 7. Outputs to Losses
\draw[flowdash] (outD.north) -- (lossD.south);
\draw[flowdash] (outM.south) -- (lossM.north);

% 8. Pseudo Label to Mask Loss
\draw[flowdash] (pseudo.east) -- (lossM.west);
\draw[flowdash] (grad_down) -| (pseudo.north);
