% ==================== NODES ====================
% Input
\node[data] (input) {9$\times$9 Light Field Data \\ $[B, 9, 9, H, W, 3]$};

% Row 1: Non-Lambertian Path (Top)
\node[innov, above right=1.2cm and 1.5cm of input.east, anchor=west] (decomp) {Three-Layer Angular Signal Decomposition \& Mask Generation Module};
\node[data, right=1.0cm of decomp] (geo) {Geometric Angular Features \\ $[B, H, W, C_{geo}]$};
\node[encoder, right=1.0cm of geo] (nl) {Non-Lambertian Geometric Branch};
\node[data, right=1.0cm of nl] (f_nl) {NL Depth Features \\ $[B, C_{nl}, H, W]$};

% Row 2: Mask \& Fusion Path (Middle)
\node[data, right=1.8cm of input] (mask) {Pixel-wise Dual-Mask \\ $[B, 2, H, W]$};
\node[innov, right=1.8cm of mask] (fusion) {Balanced Projection \& Mask-Weighted Fusion Module};
\node[decoder, right=1.2cm of fusion] (dec) {Depth Decoder};
\node[data, right=1.0cm of dec] (out) {Final Depth Map \\ $[B, 1, H, W]$};

% Row 3: Lambertian Path (Bottom)
\node[encoder, below right=1.2cm and 1.5cm of input.east, anchor=west] (lam) {Lambertian EPI Branch};
\node[data, right=1.0cm of lam] (f_lam) {Lam Depth Features \\ $[B, C_{lam}, H, W]$};

% ==================== BACKGROUND GROUPS ====================
\begin{scope}[on background layer]
  \node[fit=(decomp) (geo) (nl) (f_nl), bgbox, fill=blue!5, draw=blue!40, label=above:{\small \textbf{Non-Lambertian Path}}] (bg_nl) {};
  \node[fit=(lam) (f_lam), bgbox, fill=green!5, draw=green!40, label=below:{\small \textbf{Lambertian Path}}] (bg_lam) {};
\end{scope}

% ==================== CONNECTIONS ====================
% Input Split
\coordinate (split1) at ([xshift=0.8cm]input.east);
\draw[arrow] (input.east) -- (split1);
\draw[arrow] (split1) |- (decomp.west);
\draw[arrow] (split1) |- (lam.west);

% Decomposition to Features \& Mask

% Geometric to NL Branch

\draw[arrow] (nl) -- (f_nl);

% Lambertian Branch
\draw[arrow] (lam) -- (f_lam);

% Mask Guidance (Dashed)

\draw[dasharrow] (mask.south) -- ++(0,-0.8) -| ([yshift=-0.2cm]lam.west);
\draw[dasharrow] (mask) -- (fusion);

% Feature Fusion
\draw[arrow] (f_nl.west) -- ++(-0.6,0) |- (fusion.north);
\draw[arrow] (f_lam.west) -- ++(-0.6,0) |- (fusion.south);

% Decoder \& Output
\draw[arrow] (fusion) -- (dec);
\draw[arrow] (dec) -- (out);

% ==================== FORMULAS \& ANNOTATIONS ====================
\node[below=0.6cm of bg_lam.south west, anchor=north west, font=\scriptsize, align=left, text width=16cm, fill=white, draw=gray!30, rounded corners, inner sep=6pt] (formulas) {

  \textit{Angular Signal Decomposition:} $I(u,v) = DC + a \cdot u + b \cdot v + \varepsilon(u,v)$, \quad $Mask = \sigma(\alpha \cdot \text{angular\_gradient} + \beta)$ \\
  \textit{Mask-Weighted Fusion:} $F_{fuse} = Mask_{Lam} * Proj_{Lam}(F_{lam}) + Mask_{NL} * Proj_{NL}(F_{nl})$, \quad $Depth = Decoder(F_{fuse})$
};
